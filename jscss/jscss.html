Game Settings & Toggles

Adjust core game behaviors like weapon positioning and map lighting modes.


// --- Global Game Settings & Toggles ---

// Adjust Weapon Position (0 = down, 1 = default, 2 = high)
// Tweak the values for weaponPosition, e.g., 0.5 for slightly down.
window.weaponPosition = 0; // Current setting: Weapon down

// Set Map Lighting Mode: "golden" for golden hour, "dark" for a darker feel
// If you want default lighting, you can set this to anything else, or remove it.
window.mapMode = "golden"; // Current setting: Golden hour lighting

// Adjust Global Weapon Volume (0 to 1, 1 being 100% volume)
const weaponVolume = 0.5; // Example: 50% volume for all weapons

pc.app.on('Map:Loaded', () => {
    // Apply weapon volume on map load
    const weapons = [
        'Scar', 'Shotgun', 'Sniper', 'Tec-9', 'M4',
        'LMG', 'Desert-Eagle', 'Dagger', 'AK-47', 'AWP'
    ];
    weapons.forEach((weaponName) => {
        const weaponEntity = pc.app.root.findByName(weaponName);
        if (weaponEntity && weaponEntity.sound) {
            weaponEntity.sound.volume = weaponVolume;
        }
    });

    // Apply map lighting based on window.mapMode
    // This function needs to be defined elsewhere in your script, e.g., in save 3 UI.js
    // For manual tweaking:
    let exposure = 1.0;
    let skyboxIntensity = 1.0;
    let lightColor = { r: 1, g: 1, b: 1, a: 1 }; // Default white lighting

    if (window.mapMode === "golden") {
        exposure = 0.4; // Darker scene exposure
        skyboxIntensity = 2.0; // Brighter skybox
        lightColor = { r: 1.0, g: 0.85, b: 0.6, a: 1 }; // Golden light tint
    } else if (window.mapMode === "dark") {
        exposure = 0.1; // Very dark exposure
        skyboxIntensity = 0.5; // Dim skybox
        lightColor = { r: 0.2, g: 0.2, b: 0.2, a: 1 }; // Dark ambient light
    }

    pc.app.renderer.scene.exposure = exposure;
    pc.app.renderer.scene.skyboxIntensity = skyboxIntensity;
    const sceneLight = pc.app.root.findByName("Light");
    if (sceneLight && sceneLight.light) {
        sceneLight.light.color = lightColor;
    }

    // Apply weapon position on map load if desired
    // This is a more direct way than overriding setHeroSkin if global position is enough
    const weaponposGlobal = { x: -0.01, y: 0, z: 0.45 }; // Example: Slightly left, default height, pulled back
    if (window.weaponPosition === 0) { // Example for "weapon down"
        weaponposGlobal.y = -0.15; // Move down
        weaponposGlobal.z = 0.6; // Pull back slightly more
    } else if (window.weaponPosition === 2) { // Example for "weapon high"
        weaponposGlobal.y = 0.1; // Move up
        weaponposGlobal.z = 0.3; // Push forward slightly
    }

    // Apply to the WeaponCenter entity
    pc.app.root.findByName("WeaponCenter").setLocalPosition(
        weaponposGlobal.x,
        weaponposGlobal.y,
        weaponposGlobal.z
    );
});

Player & Character Customization

Modify character appearance, materials, and visibility of elements like hands.


// --- Character Skin Material & Emission ---
// This function overrides the game's default setHeroSkin method to add custom material properties.
// Example: Adds a white emission effect to character skins.
pc.Application.getApplication().once("postrender", function () {
    // Assuming pc.app.scripts.list()[11] is the correct script containing setHeroSkin.
    // This index might vary in different game versions.
    if (pc.app.scripts.list()[11]) {
        const originalSetHeroSkin = pc.app.scripts.list()[11].prototype.setHeroSkin;

        pc.app.scripts.list()[11].prototype.setHeroSkin = function () {
            originalSetHeroSkin.apply(this, arguments); // Call original method first

            var self = this;
            var skinName = this.heroSkin;
            var characterModel = this.characterEntity.model;

            if (characterModel && characterModel.meshInstances && characterModel.meshInstances.length > 0) {
                // Apply emission to all meshes of the character model
                for (var i = 0; i < characterModel.meshInstances.length; ++i) {
                    var mesh = characterModel.meshInstances[i];
                    if (mesh.material) {
                        // Clone the material to avoid modifying the original asset directly
                        const customMaterial = mesh.material.clone();

                        // Enable emission
                        customMaterial.emissiveFactor = 1; // Full emission strength
                        customMaterial.emissive = new pc.Color(1, 1, 1); // White emission color
                        customMaterial.update(); // Apply changes

                        mesh.material = customMaterial; // Assign the new material
                    }
                }
            }
        };
        console.log('Custom setHeroSkin override enabled for emission.');
    }
});

UI Visibility

Hide or show various user interface elements like banners, scoreboards, and health bars.


// --- Hide Specific UI Elements ---
// This function aims to hide a variety of UI elements within the game.
function applyAdditionalUIChanges() {
    try {
        const overlay = pc.app.root.findByName("Overlay");
        if (overlay) {
            // Hide Time and Score background
            const timesAndScoreBg = overlay.findByName("TimesAndScore")?.findByName("Background");
            if (timesAndScoreBg) timesAndScoreBg.enabled = false;

            // Hide 728x90 Banner (e.g., ad banner)
            const banner728x90 = overlay.findByName("728x90-Banner");
            if (banner728x90 && banner728x90.element) {
                banner728x90.element.opacity = 0; // Make it completely transparent
            }

            // Hide Objective Holder elements
            const objectiveHolder = overlay.findByName("ObjectiveHolder");
            if (objectiveHolder) {
                const objective = objectiveHolder.findByName("Objective");
                if (objective) objective.enabled = false;
                const objectiveTime = objectiveHolder.findByName("ObjectiveTime");
                if (objectiveTime) objectiveTime.enabled = false;
            }

            // Hide the actual health bar element
            const healthBar = overlay.findByName("Health");
            if (healthBar && healthBar.element) {
                healthBar.element.opacity = 0; // Make health bar invisible
            }

            // Hide Unlock Item Reward notification
            const unlockItemReward = pc.app.root.findByName("Game")?.findByName("UnlockItemReward");
            if (unlockItemReward) {
                unlockItemReward.enabled = false;
            }

            console.log("Additional UI changes applied.");
        }
    } catch (e) {
        console.error("Error applying additional UI changes:", e);
    }
}

// Call the function on map load with a delay
pc.app.on("Map:Loaded", () => {
    setTimeout(applyAdditionalUIChanges, 2000); // 2-second delay to ensure UI elements are rendered
});

    }
});

// --- Custom Reticle Implementation ---
// Creates a canvas overlay for drawing custom reticles based on the equipped weapon.
const canvas2 = document.createElement('canvas');
const ctx2 = canvas2.getContext('2d');
canvas2.width = 250; // Set the desired width for your reticle canvas
canvas2.height = 250; // Set the desired height
document.body.appendChild(canvas2);

canvas2.style.position = 'absolute';
canvas2.style.top = '50%';
canvas2.style.left = '50%';
canvas2.style.transform = 'translate(-50%, -50%)';
canvas2.style.pointerEvents = 'none'; // Important: allows mouse events to pass through

// Function to clear the reticle canvas
function clearCanvas() {
    ctx2.clearRect(0, 0, canvas2.width, canvas2.height);
}

// Example Reticle Drawing Functions (customize these!)
function drawRedDotReticle() {
    clearCanvas();
    ctx2.beginPath();
    ctx2.arc(canvas2.width / 2, canvas2.height / 2, 2, 0, Math.PI * 2, true); // Red dot
    ctx2.fillStyle = 'red';
    ctx2.fill();
    ctx2.closePath();
}

function drawACSSReticle() {
    clearCanvas();
    // Example: Simple crosshair with center dot
    ctx2.strokeStyle = 'cyan';
    ctx2.lineWidth = 1.5;

    // Center dot
    ctx2.beginPath();
    ctx2.arc(canvas2.width / 2, canvas2.height / 2, 1, 0, Math.PI * 2, true);
    ctx2.fillStyle = 'cyan';
    ctx2.fill();
    ctx2.closePath();

    // Horizontal line
    ctx2.beginPath();
    ctx2.moveTo((canvas2.width / 2) - 10, canvas2.height / 2);
    ctx2.lineTo((canvas2.width / 2) + 10, canvas2.height / 2);
    ctx2.stroke();

    // Vertical line
    ctx2.beginPath();
    ctx2.moveTo(canvas2.width / 2, (canvas2.height / 2) - 10);
    ctx2.lineTo(canvas2.width / 2, (canvas2.height / 2) + 10);
    ctx2.stroke();
}

function drawSMGReticle() {
    clearCanvas();
    // Example: Small circle
    ctx2.beginPath();
    ctx2.arc(canvas2.width / 2, canvas2.height / 2, 5, 0, Math.PI * 2, true);
    ctx2.strokeStyle = 'lime';
    ctx2.lineWidth = 1.5;
    ctx2.stroke();
    ctx2.closePath();
}

// Event listener to show/hide/change reticle based on player focus and weapon
pc.app.on('Player:Focused', (state) => {

            }
        }, 100);
    } else {
        clearCanvas(); // Clear reticle when player unfocused
    }
});

// --- Custom Headshot Hitmarker ---
// Changes the visual effect of the headshot hitmarker.
setTimeout(() => {
    // Assuming pc.app.scripts.list()[7] is the script responsible for hitmarkers.
    if (pc.app.scripts.list()[7]) {
        pc.app.scripts.list()[7].prototype.onHeadshot = function (t, e) {
            this.app.fire("DealtDamage:Trigger", e, t, true);
            this.screenEntity.sound.play("Headshot"); // Play default headshot sound
            this.hitmarkerEntity.element.opacity = 1; // Make hitmarker visible
            this.hitmarkerEntity.setLocalScale(0.3, 0.3, 0.3); // Initial small size
            this.hitmarkerEntity
                .tween(this.hitmarkerEntity.getLocalScale())
                .to(
                    {
                        x: 1.2, // Scale up to 1.2x
                        y: 1.2,
                        z: 1.2,
                    },
                    0.25,
                    pc.BackOut // Easing function for a "pop" effect
                )
                .start();
            this.lastHeadshot = Date.now();
        };
        console.log('Custom headshot hitmarker enabled.');
    }
}, 1000); // Delay to ensure script is loaded
